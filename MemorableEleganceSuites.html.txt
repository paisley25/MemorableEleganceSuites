/* Smooth scroll helper (balanced ~1.2s) */
function smoothScrollTo(targetY, duration) {
  var startY = window.scrollY || window.pageYOffset;
  var startTime = performance.now();
  var diff = targetY - startY;
  if (!duration) duration = 1200;
  function easeInOutCubic(t) { return t<0.5 ? 4*t*t*t : 1- Math.pow(-2*t+2,3)/2; }
  function step(now) {
    var elapsed = now - startTime;
    var progress = Math.min(1, elapsed / duration);
    var eased = easeInOutCubic(progress);
    window.scrollTo(0, Math.round(startY + diff * eased));
    if (elapsed < duration) requestAnimationFrame(step);
  }
  requestAnimationFrame(step);
}

/* data-scroll anchors use custom smooth scroll */
document.addEventListener('click', function(e){
  var el = e.target.closest('[data-scroll]');
  if (!el) return;
  e.preventDefault();
  var href = el.getAttribute('href') || '#';
  var id = href.replace(/^.*#/, '#');
  var target = document.querySelector(id);
  if (!target) return;
  var top = target.getBoundingClientRect().top + window.scrollY - 20;
  smoothScrollTo(top, 1200);
});

/* Add shimmer rect to SVG and animate */
function addShimmer(svg){
  var ns = "http://www.w3.org/2000/svg";
  var rect = document.createElementNS(ns,'rect');
  rect.setAttribute('x','-35%');
  rect.setAttribute('y','0');
  rect.setAttribute('width','40%');
  rect.setAttribute('height','100%');
  rect.setAttribute('fill','url(#glint)');
  rect.setAttribute('opacity','0');
  rect.setAttribute('transform','skewX(-20)');
  rect.style.transition = 'opacity 0.45s ease, transform 1.6s ease';
  svg.appendChild(rect);
  return rect;
}

/* On load: show logo, play divider shimmer; on mobile play one-time shimmer */
window.addEventListener('load', function(){
  var svg = document.getElementById('meLogo');
  if (svg) svg.classList.add('show');
  var title = document.querySelector('.title');
  if (title) title.classList.add('warm-shadow');
  var d1 = document.getElementById('div1'), d2 = document.getElementById('div2');
  if (d1) d1.classList.add('shimmer'); if (d2) d2.classList.add('shimmer');

  var isTouch = ('ontouchstart' in window) || navigator.maxTouchPoints > 0 || navigator.msMaxTouchPoints > 0;
  if (isTouch && svg){
    var r = addShimmer(svg);
    requestAnimationFrame(function(){ r.setAttribute('opacity','1'); r.setAttribute('transform','translate(240,0) skewX(-20)'); });
    setTimeout(function(){ r.setAttribute('opacity','0'); try{ svg.removeChild(r);}catch(e){} }, 1200);
  }
});

/* Hover shimmer on desktop (repeats each hover) */
(function(){
  var svg = document.getElementById('meLogo');
  if (!svg) return;
  var active = false;
  svg.addEventListener('mouseenter', function(){
    if (active) return;
    active = true;
    var rect = addShimmer(svg);
    requestAnimationFrame(function(){ rect.setAttribute('opacity','1'); rect.setAttribute('transform','translate(260,0) skewX(-20)'); });
    setTimeout(function(){ rect.setAttribute('opacity','0'); setTimeout(function(){ try{ svg.removeChild(rect);}catch(e){} active=false; }, 300); }, 1400);
  });
})();
